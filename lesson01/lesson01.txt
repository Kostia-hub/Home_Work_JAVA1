	public static void year(int year){            // задача 5
	    if(year%400 == 0){
            System.out.println("Год " + year + " - високосный");
        }else if(year%100 == 0){
            System.out.println("Год " + year + " - не високосный");
        }else if(year%4 == 0){
            System.out.println("Год " + year + " - високосный");    
        }else{
            System.out.println("Год " + year + " - не високосный");
        }
	}
